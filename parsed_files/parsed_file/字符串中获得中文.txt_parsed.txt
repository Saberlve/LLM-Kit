def is_chinese_char(char):

    unicode_char = ord(char)

    return (

        (0x4E00 <= unicode_char <= 0x9FFF) or

        (0x3400 <= unicode_char <= 0x4DBF) or

        (0x20000 <= unicode_char <= 0x2A6DF) or

        (0x2A700 <= unicode_char <= 0x2B73F) or

        (0x2B740 <= unicode_char <= 0x2B81F) or

        (0x2B820 <= unicode_char <= 0x2CEAF) or

        (0x2CEB0 <= unicode_char <= 0x2EBEF) or

        (0xF900 <= unicode_char <= 0xFAFF) or

        (0x2F800 <= unicode_char <= 0x2FA1F)

    )



def classify_characters(s):

    chinese_chars = []

    other_chars = []

    digits = []

    for char in s:

        if char.isdigit():

            digits.append(char)

        elif is_chinese_char(char):

            chinese_chars.append(char)

        else:

            other_chars.append(char)

    return chinese_chars, other_chars, digits



# 使用示例

string = '张欣然（324）'

chinese, other, nums = classify_characters(string)

a=''

for c in chinese:

    a=a+c

print("中文字符:", a)

print("其他字符:", other)

print("数字:", nums)
